选择排序递归实现

def sel_sort_rec(seq, n):
    if n == 0:
        return
    max_j = n - 1
    for j in range(n):
        if seq[j] > seq[max_j]:
            max_j = j
    seq[n-1], seq[max_j] = seq[max_j], seq[n-1]
    sel_sort_rec(seq, n-1)

选择排序循环实现

def sel_sort(seq):
    for i in range(len(seq)-1, 0, -1):
        max_j = i
        for j in range(i):
            if seq[j] > seq[max_j]:
                max_j = j
        seq[i], seq[max_j] = seq[max_j], seq[i]
    return seq

插入排序递归实现

def ins_sort_rec(seq, n):
    if n == 0:
        return
    ins_sort_rec(seq, n-1)
    j = n - 1
    while j > 0 and seq[j-1] > seq[j]:
        seq[j-1], seq[j] = seq[j], seq[j-1]
        j -= 1

def ins_sort(seq):

        j = i
        while j > 0 and seq[j-1] > seq[j]:
            seq[j-1], seq[j] = seq[j], seq[j-1]
            j -= 1

归并排序算法递归实现

def merge_sort(alist):
    if len(Alist) <= 1:
        return Alist
    middle = len(Alist) // 2
    leftA = Alist[:middle]
    rightA = Alist[middle:]
    leftA_sorted = merge_sort(leftA)
    rightA_sorted = merge_sort(rightA)
    return merge(leftA_sorted, rightA_sorted)


def merge(leftS, rightS):
    """
	合并两个已经排好序的序列
	"""
    i, j = 0, 0
    alist = []
    while i < len(leftS) and j < len(rightS):
        if leftS[i] < rightS[j]:
            alist.append(leftS[i])
            i += 1
        else:
            alist.append(rightS[j])
            j += 1
    while i < len(leftS):
        alist.append(leftS[i])
        i += 1
    while j < len(rightS):
        alist.append(rightS[j])
        j += 1
    return alist

快速排序算法

def quick_sort(alist):
    """
    快速排序（封装一层）
    :param alist: 待排序列表
    :return:
    """
    return _q_sort(alist, 0, len(alist)-1)

def _q_sort(alist, left, right):
    """
    递归实现
    :param alist: 列表
    :param left: 左边下标
    :param right: 右边下标
    :return:
    """
    if left < right:
        pivot = _partition2(alist, left, right)
        _q_sort(alist, left, pivot-1)
        _q_sort(alist, pivot+1, right)

def _partition(alist, left, right):
    """
    找到分开列表的基准
    """
    pivotion = alist[left]
    while left < right:
        while left < right and pivotion <= alist[right]:
            right -= 1
        if left < right:
            alist[left], alist[right] = alist[right], alist[left]
        while left < right and pivotion > alist[left]:
            left += 1
        if left < right:
            alist[left], alist[right] = alist[right], alist[left]
    return left

第二种实现

def _partition2(alist, left, right):
    pivotion = alist[left]
    while left < right:
        while left < right and alist[right] >= pivotion:
            right -= 1
        while left < right and alist[right] < pivotion:
            alist[left] = alist[right]
            left += 1
            alist[right] = alist[left]
    alist[left] = pivotion
    return left
